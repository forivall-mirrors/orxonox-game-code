SET ( AUDIO_SRC_FILES
  AudioBuffer.cc
  AudioManager.cc
  AudioSource.cc
  AudioStream.cc
)

ADD_CXX_FLAGS("-wd4244" MSVC)

IF (MINGW)
  ADD_LIBRARY( audio ${AUDIO_SRC_FILES} )
ELSE (MINGW)
  ADD_LIBRARY( audio SHARED ${AUDIO_SRC_FILES} )
ENDIF (MINGW)

SET_TARGET_PROPERTIES(audio PROPERTIES DEFINE_SYMBOL "AUDIO_SHARED_BUILD")
TARGET_LINK_LIBRARIES( audio
  ${OPENAL_LIBRARY}
  ${ALUT_LIBRARY}
  ${VORBISFILE_LIBRARY}
  ${VORBIS_LIBRARY}
  ${OGG_LIBRARY}
  core
  util
)

IF (NOT WIN32)
  INSTALL(TARGETS audio LIBRARY DESTINATION lib)
ENDIF (NOT WIN32)
